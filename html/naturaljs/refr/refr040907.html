1.2. 헤더고정형 그리드에 데이터 바인드(Row Handler 적용)

rowHandler 이벤트는 그리드 렌더링 시 행이 만들어질때마다 발생하는 이벤트 입니다. 기본 제공하는 그리드의 기능을 확장하기 위해 사용됩니다.

<table id="gridExample12">
	<thead><!-- 컬럼 타이틀 영역 -->
		<tr>
			<th>name</th>
			<th>age</th>
			<th>email</th>
			<th>registered</th>
		</tr>
	</thead>
	<tbody><!-- 데이터가 반복되면서 바인드 되는 부분 -->
		<tr>
			<td><input id="name" type="text" /></td>
			<td><input id="age" type="text" /></td>
			<td><input id="email" type="text" /></td>
			<td id="registered"></td>
		</tr>
	</tbody>
</table>

<script type="text/javascript">
	N(data).grid({
		context : "#gridExample12",
		height : 200,
		rowHandler : function(index, row, data) {
			// age 가 28 이상이면 해당 셀의 배경색을 red 로 설정
			if(data.age >= 28) {
				row.find("#age").css("background-color", "red");
			}
		}
	}).bind();
</script>

2. 신규 행 추가

<table id="gridExample20">
	<thead><!-- 컬럼 타이틀 영역 -->
		<tr>
			<th>name</th>
			<th>age</th>

		<tr>
			<td><input id="name" type="text" /></td>
			<td><input id="age" type="text" data-validate='[ ["required"], ["integer"] ]' /></td> <!-- 필수입력과 숫자만 입력 체크 -->
			<td><input id="email" type="text" data-validate='[ ["email"] ]' /></td> <!-- 이메일 주소 체크 -->
			<td><input id="registered" type="text" data-format='[ ["date", 8, "date"] ]' /></td> <!-- 날자포멧 및 데이트피커 적용 -->
		</tr>
	</tbody>
</table>
<button id="btnGridExample20">행추가</button>

<script type="text/javascript">
	var grid = N(data).grid("#gridExample20").bind();

	// 행추가
	N("#btnGridExample20").click(function() {
		grid.add(); //add 메서드를 호출하면 id 속성을 가지고 있는 입력요소를 기반으로 { "age": "", "email": "", "registered" : "" } 와 같은 데이터가 추가됨.
	});
</script>

<button id="btnGridExample30">저장</button>

<script type="text/javascript">
	var grid = N(data).grid("#gridExample30").bind();

	N("#btnGridExample30").click(function() {
		N(window).alert({ // 1. confirm 메시지 출력
			msg : "저장 하겠습니까?",
			confirm : true,
			onOk : function() {
				if(grid.validate()) { // 2. 그리드에 입력된 데이터 검증
					N(grid.data()).comm("save.json").submit(); // 3. 그리드의 데이터를 저장
				}
			}
		});
	});
</script>

4. N.select 와 N.grid 연동

N.select 와 N.grid을 연동하여 사용할 때는 반드시 N.select 를 먼저 초기화 하고 N.grid 를 그 다음에 초기화 해야 합니다.

<table id="gridExample40">
	<thead><!-- 컬럼 타이틀 영역 -->
		<tr>
			<th>name</th>
			<th>age</th>
			<th>gender</th>
			<th>email</th>
			<th>registered</th>
		</tr>
	</thead>
	<tbody><!-- 데이터가 반복되면서 바인드 되는 부분 -->
		<tr>
			<td><input id="name" type="text" /></td>
			<td><input id="age" type="text" /></td>
			<td><input id="gender" type="radio" /></td>
			<td><input id="email" type="text" /></td>
			<td id="registered"></td>
		</tr>
	</tbody>
</table>

<script type="text/javascript">
	// 1. gender 라디오 셀렉트 생성
	N([{
     	"key" : "male",
     	"val" : "male"
     },{
     	"key" : "female",
     	"val" : "female"
    }]).select({
    	context : "#gridExample40 #gender",
    	key : "key",
    	val : "val"
    })

    // 2. 그리드 바인드
	N(data).grid("#gridExample40").bind();
</script>

5. onSelect 이벤트를 이용하지 않고 행 의 두번째 셀 클릭 시 동작하는 이벤트 핸들러를 jQuery on 을 이용하여 직접 정의.

N.grid 의 행 요소들은 다시 바인드 하면 사라지고 다시 생성 되므로 bind나 click 대신 on 메서드를 이용하여 이벤트를 바인드 해야 합니다.

// Controller
N("selector").cont(function() {
    // 1. 그리드 바인드
	var grid = N(data).grid(N("selector"), view).bind();

    // 2. 행의 두번째 셀 클릭 시 동작하는 이벤트 핸들러 정의
	grid.context("tbody").on("click", "td:eq(1)", function() {

		// 3. 행 인덱스 얻어오기
		var rowEle = $(this).closest("tbody");
		var idx = grid.context("tbody").index(rowEle);
		var rowData = grid.data()[idx];

		// 4. 가져 온 행 인덱스와 행 요소, 행 데이터를 이용하여 로직을 처리
		N.log(idx, rowEle, rowData);
	});
});
