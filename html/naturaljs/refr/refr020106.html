1. Define View and Controller

<!-- View area start -->
<article class="view">
    <p>View</p>
</article>
<!-- View area end -->

<script type="text/javascript">
    // Controller area start
    N(".view").cont({
        init : function(view, request) {
            N("p", view).css("padding", "10px");
        }
    });
    // Controller area end
</script>
